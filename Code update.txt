const generateImageWithOpenAI = async (prompt, subjectName) => {
    const response = await fetch('https://api.openai.com/v1/images/generations', {
        headers: {
            'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
            'Content-Type': 'application/json',
        },
        method: 'POST',
        body: JSON.stringify({
            prompt: `${prompt} - Jimbo Phillips skateboard art style with vibrant neon colors, monster elements, bold outlines`,
            n: 1,
            size: "1024x1792", // Tall for skateboard
            quality: "hd",
            style: "vivid"
        }),
    });

    if (!response.ok) {
        throw new Error(`OpenAI API error: ${response.status}`);
    }

    const result = await response.json();
    return result.data[0].url;
};
